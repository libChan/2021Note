class Solution {
    public int fib(int n) {
        if(n == 0) return 0;
        else if(n == 1)return 1;
        int a = 0;
        int b = 1;
        int sum = 0;
        for(int i = 2;i<n+1;i++){
            sum = (a+b) % 1000000007;
            a = b;
            b = sum;
        }
        return b;
    }
}
