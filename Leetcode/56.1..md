class Solution {
    public int[] singleNumbers(int[] nums) {
        int sum = 0;
        for(int n : nums){
            sum ^= n;
        }
        int mask = 1;
        while((sum & mask) == 0){
            mask <<= 1;
        }
        int a = 0, b = 0;
        for(int n : nums){
            if((n & mask) == 0) a ^= n;
            else b ^= n;
        }
        return new int[]{a,b};
    }
}
