class Solution {
    int dfs(TreeNode node){
        if(node == null) return 0;
        int left = dfs(node.left);
        if(left == -1) return -1;
        int right = dfs(node.right);
        if(right == -1) return -1;
        return Math.abs(left-right) > 1 ? -1 : Math.max(left, right)+1;
    }

    public boolean isBalanced(TreeNode root) {
        return dfs(root) != -1;
    }
}
