```java
class Solution {
    int maxLen = 0;
    int deep = 0;
    void dfs(TreeNode node){
        if(node != null){
            deep++;
            dfs(node.left);
            dfs(node.right);
            deep--;
        }
        if(deep > maxLen) {
            maxLen = deep;
        }
    }
    public int maxDepth(TreeNode root) {
        dfs(root);
        return maxLen;
    }
}	
```

```java
class Solution {
    public int maxDepth(TreeNode root) {
        if(root == null) return 0;
        return Math.max(maxDepth(root.left), maxDepth(root.right)) + 1;
    }
}
```
